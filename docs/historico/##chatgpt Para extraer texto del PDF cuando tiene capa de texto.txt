 ## para extraer texto del PDF cuando tiene capa de texto, el código base es este:

import re

try:
    import fitz  # PyMuPDF
except ImportError:
    fitz = None


def extract_text_from_pdf(pdf_path):
    """
    Extrae texto de un PDF con PyMuPDF.
    En PDFs escaneados o con columnas, hay que revisar visualmente.
    """
    if fitz is None:
        raise RuntimeError("Instala pymupdf: pip install pymupdf")

    doc = fitz.open(pdf_path)
    pages = []

    for page_number, page in enumerate(doc, start=1):
        text = page.get_text("text")
        pages.append(f"\n--- PAGE {page_number} ---\n{text}")

    return "\n".join(pages)


def clean_text(text):
    """
    Limpieza mínima sin reformular ni resumir.
    """
    text = text.replace("\r", "\n")
    text = re.sub(r"[ \t]+", " ", text)
    text = re.sub(r"\n{3,}", "\n\n", text)
    return text.strip()


def parse_answer_key(text):
    """
    Extrae respuestas tipo:
    1 B
    2 D
    3 A

    Conserva respuestas dudosas como D?, ANULADA o /.
    """
    answers = {}

    pattern = re.compile(
        r"(?m)^\s*(\d{1,3})\s*[\.\-:]?\s*([A-Da-d](?:\?)?|ANULADA|anulada|/)\s*$"
    )

    for match in pattern.finditer(text):
        number = int(match.group(1))
        answer = match.group(2).strip()
        answers[number] = answer

    return answers
